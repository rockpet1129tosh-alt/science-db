\begin{enumerate}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_1 + z_1 = 0
            \qquad \text{∴} \; x_1 = z_1 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{z_1}{C} + \frac{x_1}{C}
            \qquad \text{∴} \; x_1 = z_1 = \uwave{\frac{1}{2}CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_1 - y_1+ z^*_1 = 0
            \qquad \text{∴} \; y_1 - z^*_1 = -\frac{1}{2}CV_0  \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{z^*_1}{C} + \frac{y_1}{C}
            \qquad \text{∴} \; y_1 = \uwave{\frac{1}{4}CV_0} \;,
            \quad z^*_1 = \uwave{\frac{3}{4}CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_2 - y_1 + z_2 = 0
            \qquad \text{∴} \; x_2 - z_2 = -\frac{1}{4}CV_0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{z_2}{C} + \frac{x_2}{C}
            \qquad \text{∴} \; x_2 = \uwave{\frac{3}{8}CV_0} \;,
            \quad z_2 = \uwave{\frac{5}{8}CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_2 - y_2 + z^*_2 = 0
            \qquad \text{∴} \; y_2 - z^*_2 = -\frac{3}{8}CV_0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{z^*_2}{C} + \frac{y_2}{C}
            \qquad \text{∴} \; y_2 = \uwave{\frac{5}{16}CV_0} \;,
            \quad z^*_2 = \uwave{\frac{11}{16}CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -X - Y + Z = 0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            &V_0 - \frac{X}{C} = V_0 - \frac{Y}{C} = \frac{Z}{C} \\
            & \text{∴} \; X = Y = \uwave{\frac{1}{3}CV_0} \;,
            \quad Z = \uwave{\frac{2}{3}CV_0} \;. 
        \end{align*}
        
\end{enumerate}
