
\documentclass[border=1pt]{standalone}

\usepackage{tikz}

\usetikzlibrary{arrows.meta}

\usepackage{circuitikz}

\ctikzset{resistors/width=1.1, batteries/scale=1.2, 
    capacitors/width=0.2, capacitors/height=0.9,
    grounds/scale=1.6}
\ctikzset{bipoles/cuteswitch/thickness=0.23}

\begin{document}
\begin{circuitikz}

    \pgfmathsetmacro{\x}{6.9}
    \pgfmathsetmacro{\xa}{3.2}
    \pgfmathsetmacro{\w}{2.8}
    \pgfmathsetmacro{\t}{0.1}
    \pgfmathsetmacro{\y}{4}
    \pgfmathsetmacro{\h}{0.18}
    \pgfmathsetmacro{\d}{1.2}

    \draw (0,0) to [battery1, invert, l_=B] ++ (0,\y)
    to [cute open switch, l=S$_1$, -*] ++ (\xa,0) coordinate (A)
    to [cute open switch, l=S$_2$] ++ (\x-\xa,0)
    to [short] ++ (0,-\y/2+\h);
    \draw (\x,\y/2-\d-\h) -- (\x, 0)
    to[short, -*] (\xa,0) coordinate (B)
    to[short] (0,0) node [ground] {};

    \draw[thick] (\xa-\w/2,\y/2+\d+\h) rectangle ++ (\w,-\h)
        node [right=-1.5mm,below=-0.3mm]{X} ;
    \draw[thick] (\xa-\w/2,\y/2-\d-\h) rectangle ++ (\w,+\h)
        node[right=-1.5mm,above=-0.3mm]{Y} ;        
    \draw[thick] (\x-\w/2,\y/2+\h) rectangle ++ (\w,-\h)
        node [right=-1.5mm,below=-0.3mm]{Z} ;
    \draw[thick] (\x-\w/2,\y/2-\d-\h) rectangle ++ (\w,+\h)
        node[right=-1.5mm,above=-0.3mm]{W} ;
        
    \draw (A) to [short] ++ (0,-\y/2+\d+\h);
    \draw (B) to [short] ++ (0,\y/2-\d-\h);

    \draw[semithick,<->,>={Stealth}] (\xa-\w/2+\t,\y/2-\d) -- (\xa-\w/2+\t,\y/2+\d)
        node[midway,left] {$2d$};
    \draw[semithick,<->,>={Stealth}] (\x-\w/2+\t,\y/2-\d) -- (\x-\w/2+\t,\y/2)
        node[midway,left] {$d$};

\end{circuitikz}

\end{document}
