キルヒホッフ則より，
%
\begin{align*}
    \frac{Q}{C} = RI \;. \myleader{1.8cm} \text{①}
\end{align*}
%
コンデンサの上側極板へ単位時間あたりに流出する電荷が$I$であるから，
\begin{align*}
    \dv{Q}{t} = -I \;. \myleader{1.8cm} \text{②}
\end{align*}
%

\begin{enumerate}

    \item $Q = kQ_0$のとき，①より，
        %
        \begin{align*}
            I = k \frac{Q_0}{RC} \;.
        \end{align*}
        
    \item ①，②より，
        %
        \begin{align*}
            \dv{Q}{t} = - \frac{1}{RC} Q \;.
        \end{align*}
        %
        よって，
        %
        \begin{align*}
            Q = \uwave{Q_0 e^{-t/RC}} \;,
            \quad I = \uwave{\frac{Q_0}{RC} e^{-t/RC}} \;.
        \end{align*}

\end{enumerate}
