\begin{enumerate}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_1 + y_1 = 0 \qquad \text{∴} \; x_1 = y_1 \;. 
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{y_1}{2C} +  \frac{x_1}{C}
            \qquad \text{∴} \; x_1 = y_1 = \uwave{\frac{2}{3} CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_1 + y^*_1 + z_1 = 0
            \qquad \text{∴} \; y^*_1 + z_1 = \frac{2}{3} CV_0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            \frac{y^*_1}{2C} = \frac{z_1}{2C} 
            \qquad \text{∴} \; y^*_1 = z_1 = \uwave{\frac{1}{3} CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_2 + y_2 + z_1 = 0
            \qquad \text{∴} \; x_2 - y_2 = \frac{1}{3} CV_0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            V_0 = \frac{y_2}{2C} +  \frac{x_2}{C}
            \qquad \text{∴} \; x_2 = \uwave{\frac{7}{9} CV_0} \;,
            \quad y_2 = \uwave{\frac{4}{9} CV_0} \;.
        \end{align*}

    \item 電荷保存則より，
        %
        \begin{align*}
            -x_2 + y^*_2 + z_2 = 0
            \qquad \text{∴} \; y^*_2 + z_2 = \frac{7}{9} CV_0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            \frac{y^*_2}{2C} = \frac{z_2}{2C} 
            \qquad \text{∴} \; y^*_2 = z_2 = \uwave{\frac{7}{18} CV_0} \;.
        \end{align*}

    \item C$_1$，C$_2$，C$_3$の電荷を$X$，$Y$，$Z$とおく．電荷保存則より，
        %
        \begin{align*}
            -X + Y + Z = 0 \;.
        \end{align*}
        %
        キルヒホッフ則より，
        %
        \begin{align*}
            \left\{
            \begin{aligned}
                & V_0 = \frac{Y}{2C} +  \frac{X}{C} \;, \\
                & \frac{Y}{2C} = \frac{Z}{2C}
            \end{aligned}
            \right.
        \end{align*}
        %
        \begin{align*}
                \text{∴} \; X = \frac{4}{5} CV_0 \;,
                \quad Y = Z = \frac{2}{5} CV_0 \;.
        \end{align*}
        %
        よって，
        %
        \begin{align*}
            V = \frac{Z}{2C} = \uwave{\frac{1}{5} V_0} \;.
        \end{align*}        

\end{enumerate}
