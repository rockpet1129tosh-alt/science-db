
\documentclass[border=1pt]{standalone}

\usepackage{tikz}

\usetikzlibrary{arrows.meta}

\usepackage{circuitikz}

\ctikzset{resistors/width=1.1, batteries/scale=1.2, 
    capacitors/width=0.2, capacitors/height=0.9,
    grounds/scale=1.6}
\ctikzset{bipoles/cuteswitch/thickness=0.23}

\begin{document}

\begin{circuitikz}

    \pgfmathsetmacro{\w}{5.1}
    \pgfmathsetmacro{\t}{0.1}
    \pgfmathsetmacro{\h}{0.2}
    \pgfmathsetmacro{\d}{1.3}

    \draw[thick] (0,0) rectangle (\w,\h) node[right=-1.5mm,above=-0.3mm] {X};
    \draw[thick] (0,-\d-\h) rectangle (\w,-\d) node[right=-1.5mm,above=-0.3mm] {Y};
    \draw[thick] (0,-3*\d-2*\h) rectangle (\w,-3*\d-\h) node[right=-1.5mm,above=-0.3mm] {Z};
    \node[ground] at (4/5*\w,-3*\d-2*\h) {};

    \draw[semithick,<->,>={Stealth}] (\t,0) -- (\t,-\d)
        node[midway,left] {$d$};
    \draw[semithick,<->,>={Stealth}] (\t,-\d-\h) -- (\t,-3*\d-\h)
        node[midway,left] {$2d$};

\end{circuitikz}

\end{document}
