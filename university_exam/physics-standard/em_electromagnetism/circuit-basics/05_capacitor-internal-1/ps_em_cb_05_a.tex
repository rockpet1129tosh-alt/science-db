            -x_1 + y_1 = 0 \qquad \text{∴} \; y_1 = \uwave{Q_0} \;.
        \end{align*}
        
    \item Yについての電荷保存則より，

        \end{align*}

        %
        \begin{align*}
            \frac{y_2}{\varepsilon_0 S} \cdot 2d +

            = \uwave{\frac{\varepsilon_0 S}{3d} V_0} \;.
        \end{align*}

        \begin{align*}
            -x_3 + y_3 = Q_0 \;.
        \end{align*}

        キルヒホッフ則より，
        %
        \begin{align*}
            & \frac{y_3}{\varepsilon_0 S} \cdot 2d + 
            \frac{x_3}{\varepsilon_0 S} \cdot d = 0 \\

            \quad y_3 = \uwave{\frac{1}{3} Q_0} \;.
        \end{align*}

\end{enumerate}
