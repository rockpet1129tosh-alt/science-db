\begin{enumerate}

    \item 力学的エネルギー保存則より，
        \begin{align*}
            \frac{1}{2}mv^2 + qEx = \frac{1}{2}m{v_0}^2
            \qquad \text{∴} \;\; |v|= \uwave{\sqrt{{v_0}^2 - \frac{2qE}{m}x}} \;.
        \end{align*}
        
    \item 位置$x$における電位は，適当な定数$\phi_0$を用いて，
        \begin{align*}
            \phi(x) = Ex + \phi_0
        \end{align*}
        %
        と書ける．基準の位置$x=h$で$\phi=0$となるためには，
        $\phi_0 = -Eh$と定めればよく，
        %
        \begin{align*}
            \phi(x) = \uwave{E(x - h)} \;.
        \end{align*}

\end{enumerate}
